#Core the Land, discard other cores on the land
country_event = {
	id = ADX_annex_country.2
	title = ADX_annex_country.2.t
	desc = ADX_annex_country.2.d

	picture = ADX_core_the_land-456
	fire_only_once = yes
	trigger = {
		tag = ADX 
		owns_state = 659
		}
	mean_time_to_happen = { days = 7 }
	timeout_days = 13
	fire_for_sender = yes
	major = yes
	option = {
		name = ADX_annex_country.2.a
		ai_chance = { factor = 1 }
		every_state = {
           limit = {
                AND = {
                   is_controlled_by = ADX
                   NOT = { is_core_of = ADX }
               }
          }
        add_core_of = ADX
		}
		        every_state = {
            limit = {
                is_core_of = ADX
                is_controlled_by = ADX
                is_owned_by = ADX
            }
            every_country = {
			limit = {
                    NOT = { has_country_flag = TEST }
                }
                remove_state_core = PREV
                remove_state_claim = PREV
            }
        }
		hidden_effect = {
			
		}
	}
}
